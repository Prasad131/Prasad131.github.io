Solving similarity equations

import pandas as pd
import numpy as np 
import matplotlib.pyplot as plt 
import scipy as sp
def make_rhs(Pr):
    def _rhs(t,x):
        f0,f1,f2,t0,t1 = x 
        return [
            f1,
            f2,
            t0-(1/Pr)*(0.5*f1**2-0.75*f0*f2),
            t1,
            0.75*f0*t1]
    return _rhs

def make_root_func(Pr):
    def _root_func(x):
        x0 = [0,0,x[0],1,x[1]]
        sol = sp.integrate.solve_ivp(make_rhs(Pr),[0,10],x0)
        if sol.success:
            return [sol.y[1,-1],sol.y[3,-1]]
        else:
            raise AssertionError("Did not converge!")
    return _root_func

def Nu_local(Pr,Ra,x0=None):
    if x0 is None:
        x0 = [-0.879,-0.386]
    sol = sp.optimize.root(make_root_func(Pr),x0)
    if sol.success:
        return -(4/3)*Ra**(1/4)*sol.x[1]
    else:
        raise AssertionError('Root did not converge!')

def Nu_mean(Pr,Ra,x0=None):
    if x0 is None:
        x0 = [-0.879,-0.386]
    Nu_l = Nu_local(Pr,Ra,x0)
    return 4*Nu_l/3

air_props = pd.read_csv('air_props.csv')

air_props.t = air_props.t+273.15

Problem 1

A vertical wall of an oven is 60 cm long and is covered with a metal sheet maintained at 170°C. Air temperature inside the oven is 90°C and the pressure is atmospheric. Calculate the local heat-transfer coefficient at the end of the wall and the average heat transfer coefficient over the wall.

L = 0.6 # m 
T_wall = 170+273.15 # K
T_infty = 90+273.15 # K
g = 9.81 # m/s^2
# determine the film temperature 
T_film = 0.5*(T_wall+T_infty)
# Compute properties at film temperature using linear interpolation from the data
beta = np.interp(T_film,air_props.t,air_props.beta)
nu = np.interp(T_film,air_props.t,air_props.nu)
alpha = np.interp(T_film,air_props.t,air_props.alpha)
Pr = np.interp(T_film,air_props.t,air_props.Pr)
Ra = (g*beta*(T_wall-T_infty)*L**3)/(nu*alpha)
kappa = np.interp(T_film,air_props.t,air_props.k)
# check if the flow is laminar
if Ra>1e09:
    raise ValueError("Turbulent regime detected!!!")
# compute h from Nu 
Nu_L = Nu_local(Pr,Ra)
h_L = Nu_L*kappa/L
Nu_avg = Nu_mean(Pr,Ra)
h_avg = Nu_avg*kappa/L 
# display the answers
print("local heat transfer coefficient = {:.2f} W/m^2K".format(h_L))
print("mean heat transfer coefficient = {:.2f} W/m^2K".format(h_avg))
