Pr=να,Ec=U2cp(TH−To),Br=PrEc Pr = \frac{\nu}{\alpha}, \quad Ec = \frac{U^{2}}{c_{p}(T_{H}-T_{o})}, \quad Br=Pr Ec

Nuo=−2(1+EcPr2),NuH=2(1−EcPr2)T(y)−ToTH−To=yH+EcPr2(yH−y2H2) \begin{align} Nu_o &= -2\left( 1+\frac{Ec Pr}{2}\right), \quad Nu_H = 2\left( 1- \frac{Ec Pr}{2} \right) \\ \frac{T(y)-T_o}{T_H-T_o} &= \frac{y}{H} + \frac{Ec Pr}{2}\left(\frac{y}{H} - \frac{y^2}{H^2} \right) \end{align}
